\frame
{
  \frametitle{Alternatives to Barker Sequence}

    \begin{itemize}
        \item Barker sequences possess minimal cross-correlation except at specific instances 
        \item There exists two alternative sequences used in wireless implementations
        \begin{itemize}
         \item Zadoff-Chu Sequences
         \item Golay Complementary Sequences
        \end{itemize}
    \end{itemize}

}

\frame
{
  \frametitle{Zadoff-Chu Sequences}

    \begin{itemize}
        \item Created by Solomon Zadoff and David Chu 
        \item Sequence numbers generated using the following expression:
         \begin{equation}
        \begin{split}
            s_n=e^{-j\pi kn(n+1+2q)/L} 
        \end{split}
        \end{equation}
        where $L$ is sequence length, $n$ is sequence index, $q$ is integer, and $k$ which is coprime with $L$
    \end{itemize}

}

\frame
{
  \frametitle{Applications of Zandoff-Chu Sequences}
    \begin{itemize}
        \item Used for LTE synchronization and channel sounding 
         \item Possess constant amplitude, zero circular auto-correlation, and low cross cross-correlation 
        \item Extremely useful in multi-access environment with multiple user transmit signals
    \end{itemize}
}

\frame
{
  \frametitle{Golay Complementary Sequences (GCS)}

    \begin{itemize}
        \item Sequences of bipolar symbols with minimal autocorrelation properties
        \item Similar to concept of Barker codes 
        \item Sequences occur in complimentary pairs
        
    \end{itemize}

}

\frame
{
  \frametitle{Golay Complementary Sequences (GCS)}

    \begin{itemize}
        \item We typically denote the pairs as $G_{an}$ and $G_{bn}$ where $n$ is sequence length 
        \item IEEE~802.11ad uses $Ga_{32}$,$Ga_{64}$ and $Gb_{64}$
    \end{itemize}

}
\frame
{
  \frametitle{GCS Advantages}

    \begin{itemize}
       \item With BPSK performance is exceptional even with severe phase rotation  
        \item Auto-correlation for sequence pairs can be performed in parallel 
        \item Preamble in IEEE~802.11ad can provide signal information 
    \end{itemize}

}
